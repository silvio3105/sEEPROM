47uint8_t sEEPROM::read(uint16_t startOffset, void* output, uint16_t len)
48{
49 // If required number of bytes to read go outside EEPROM sector
50 if ((start + startOffset + len) < (start + len)) return SEEPROM_OF;
51
52 uint16_t idx = 0;
53 uint8_t* addr = (uint8_t*)(start + startOffset);
54
55 do
56 {
57 // Read value from EEPROM
58 ((uint8_t*)output)[idx] = addr[idx];
59
60 // Increase index
61 idx++;
62 }
63 while (idx != len);
64
65 return SEEPROM_OK;
66}

68uint8_t sEEPROM::write(uint16_t startOffset, void* value, uint16_t len)
69{
70 // If required number of bytes to write go outside EEPROM sector
71 if ((start + startOffset + len) < (start + len)) return SEEPROM_OF;
72
73 // Unlock EEPROM write access
74 unlockEEPROM();
75
76 // Calculate number of 4 byte values
77 uint16_t len4 = len / 4;
78 len = len % 4;
79
80 // Calculate number of 2 byte values
81 uint16_t len2 = len / 2;
82 len = len % 2;
83
84 // Write 4 bytes values if needed
85 if (len4)
86 {
87 write<uint32_t>((uint32_t*)(start + startOffset), (uint32_t*)value, len4);
88
89 // Move offset address
90 startOffset += (len4 * 4);
91 }
92
93 // Write 2 byte values if needed
94 if (len2)
95 {
96 write<uint16_t>((uint16_t*)(start + startOffset), (uint16_t*)((uint16_t*)value + startOffset), len2);
97
98 // Move offset address
99 startOffset += (len2 * 2);
100 }
101
102 // Write 1 byte values if needed
103 if (len) write<uint8_t>((uint8_t*)(start + startOffset), (uint8_t*)((uint8_t*)value + startOffset), len);
104
105 // Lock EEPROM write access
106 lockEEPROM();
107
108 return SEEPROM_OK;
109}

111uint8_t sEEPROM::erase(uint16_t startOffset, uint16_t len)
112{
113 // Check if offset address is aligned by 4 bytes
114 if (startOffset % 4) return SEEPROM_NOK;
115
116 // Check for EEPROM overflow
117 if ((start + startOffset + (len * 4)) < (start + (len * 4))) return SEEPROM_OF;
118
119 uint16_t idx = 0;
120 uint32_t* addr = (uint32_t*)(start + startOffset);
121
122 // Unlock EEPROM write access
123 unlockEEPROM();
124
125 // Enable EEPROM erase
126 FLASH->PECR |= FLASH_PECR_ERASE;
127
128 do
129 {
130 // Erase four bytes
131 addr[idx] = 0x00;
132
133 // Wait for interrupt
134 __WFI();
135
136 // Increase index
137 idx++;
138 }
139 while (idx != len);
140
141 // Disable EEPROM erase
142 FLASH->PECR &= ~FLASH_PECR_ERASE;
143
144 // Lock EEPROM write access
145 lockEEPROM();
146
147 return SEEPROM_OK;
148}
