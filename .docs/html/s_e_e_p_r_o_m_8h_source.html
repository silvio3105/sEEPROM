148 void write(T* startAddr, T* value, uint16_t len)
149 {
150 uint16_t idx = 0;
151
152 do
153 {
154 // Wait for EEPROM if busy
155 while (FLASH->SR & FLASH_SR_BSY);
156
157 // Write value
158 startAddr[idx] = value[idx];
159
160 // Increase index
161 idx++;
162 }
163 while (idx != len);
164 }

171 inline void unlockEEPROM(void)
172 {
173 // Wait if flash is busy

175
